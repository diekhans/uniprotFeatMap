** add new swissprot accession to acc.tsv
** obtain data 
./bin/getTestData acc.tsv tmp
cp tmp/*. input/
** update alignemnts 
make -j 10 -k aligntest >&log
cp output/*.{psl,tsv,bed} expected/
** repeat until we have all updated (3 times)
make -j 10 -k test >&log
cp output/*.{psl,tsv,bed} expected/
